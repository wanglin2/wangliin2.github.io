# MarkerImgMobile 带标注的图片预览组件

# 安装

$ npm i markerimgmobile

# 引入

import MarkerImgMobile from 'markerimgmobile'
Vue.use(MarkerImgMobile)

# 基础用法

基础用法
<template>
  <div class="container">
    <MarkerImgMobile 
        :src="imgUrl" 
        :info="imgMarks" 
        :prevent="false" 
        :useWrapSize="true"
    ></MarkerImgMobile>
  </div>
</template>

<script>
export default {
  data () {
    return {
      imgUrl: require('../../../.vuepress/public/94fb997b-f094-473a-b5fd-c0d3520f742f.png'),
      imgMarks: [
        {
          ratio: false,
          xmin: 278,
          xmax: 278+313,
          ymin: 191,
          ymax: 191+357,
          text: '我是文字',
          color: '#FA3239',
          fillColor: 'rgba(250,50,57,0.30)',
          fontColor: '#fff'
        }
      ]
    }
  }
}
</script>

<style lang="less" scoped>
.container {
    width: 375px;
    height: 667px;
    background-color: #f5f5f5;
}
</style>

# API

# 属性

参数说明类型可选值默认值
src图片URLString—
info图片上线框及文字信息Array—[]
fullScreenCanDrag全屏下是否可拖动Boolean—true
fullScreenCanScale全屏下是否可缩放Boolean—true
prevent是否阻止触摸事件的默认事件Boolean—false
useWrapSize默认是获取窗口可视宽高，设为true后使用外层容器宽高Boolean—false
