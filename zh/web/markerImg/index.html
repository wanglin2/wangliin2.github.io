# MarkerImg 带标注的图片显示组件

# 安装

$ npm i markerimg

# 引入

import MarkerImg from 'markerimg'
Vue.use(MarkerImg)

# 基础用法

基础用法
<template>
<div class="container">
  <MarkerImg :src="imgUrl" :mark="imgMarks"></MarkerImg>
</div>
</template>

<script>
export default {
  data () {
    return {
      imgUrl: require('../../../.vuepress/public/94fb997b-f094-473a-b5fd-c0d3520f742f.png'),
      imgMarks: [
        {
          ratio: false,
          xmin: 278,
          xmax: 278+313,
          ymin: 191,
          ymax: 191+357,
          text: '我是文字',
          color: '#FA3239',
          fillColor: 'rgba(250,50,57,0.30)',
          fontColor: '#fff'
        }
      ]
    }
  }
}
</script>

<style lang="less" scoped>
.container {
  width: 300px;
  height: 200px;
}
</style>

# 异常情况

异常情况
<template>
<div class="container">
  <MarkerImg :src="'http://xx.com/xx.jpg'" :mark="imgMarks"></MarkerImg>
</div>
<div class="container">
  <MarkerImg :src="''" :mark="imgMarks"></MarkerImg>
</div>
</template>

<script>
export default {
  data () {
    return {
      imgMarks: []
    }
  }
}
</script>

<style lang="less" scoped>
.container {
  width: 300px;
  height: 200px;
  margin-bottom: 10px;
}
</style>

# API

# 属性

参数说明类型可选值默认值
src图片urlString—
mark图片标注，支持显示多个标注Array—[]
disableDrag是否不允许拖动Boolean—false
disableScale是否不允许缩放Boolean—false
defaultImg没有图片时的默认图String—
failImg图片加载失败的默认图String—
defaultClass默认图情况下自定义类名，添加到最外层元素上，可用来自定义样式String—
errorImgStyle错误图片的样式Object—{}
blurBg是否显示虚化背景Boolean—默认为纯黑背景
blurNum虚化度Number—15
