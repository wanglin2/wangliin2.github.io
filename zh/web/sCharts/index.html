# sCharts 图表组件

# 安装

$ npm i s-charts

# 引入

import sCharts from 's-charts'
Vue.use(sCharts)

# 使用ECharts图表

对ECharts进行简单的封装，主要是增加了对窗口尺寸变化事件的监听处理。

基础用法
<template>
  <div class="chartContent">
    <EChart :options="chartOptions"></EChart>
  </div>
</template>

<script>
const randomNum = (num = 10) => {
  return Math.ceil(num * Math.random())
}

export default {
  data () {
    return {
      chartOptions: {},
      data: (new Array(10)).fill(0).map((item, index) => {
      return {
        'regionName': '桂苑社区' + index,
        'lineChartDtoList': [
          {
            'timePoint': '周一',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          },
          {
            'timePoint': '周二',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          },
          {
            'timePoint': '周三',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          },
          {
            'timePoint': '周四',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          },
          {
            'timePoint': '周五',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          },
          {
            'timePoint': '周六',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          },
          {
            'timePoint': '周日',
            'eventNum': randomNum(),
            'dealNum': randomNum(),
            'probability': randomNum()
          }
        ]
      }
    })
    }
  },
  mounted () {
    this.createChartOptions()
  },
  methods: {
    createChartOptions () {
      let xData = this.data[0].lineChartDtoList.map(item => {
        return item.timePoint
      })
      let yData = [
        ...this.data.map(item => {
          return {
            name: item.regionName,
            type: 'line',
            data: item.lineChartDtoList.map(item2 => {
              return Number(item2.probability)
            })
          }
        })
      ]
      let legendData = this.data.map(item => {
        return item.regionName
      })
      let options = {
        tooltip: {
          trigger: 'axis',
          appendToBody: true
        },
        legend: {
          type: 'scroll',
          width: '70%',
          data: legendData,
          right: 40,
          textStyle: {
            color: 'rgba(0,0,0,0.70)',
            fontSize: 14
          }
        },
        grid: {
          left: '3%',
          right: '4%',
          bottom: '3%',
          containLabel: true
        },
        xAxis: {
          type: 'category',
          boundaryGap: false,
          data: xData,
          axisLine: {
            lineStyle: {
              color: '#CCCCCC'
            }
          },
          axisTick: {
            show: false
          },
          axisLabel: {
            interval: 0,
            fontSize: 14,
            color: 'rgba(0,0,0,0.70)',
            lineHeight: 20
          }
        },
        yAxis: {
          type: 'value',
          splitLine: {
            show: true,
            lineStyle: {
              color: 'rgba(0,0,0,0.06)'
            }
          },
          axisLine: {
            lineStyle: {
              color: '#CCCCCC'
            }
          },
          axisTick: {
            show: false
          },
          axisLabel: {
            fontSize: 14,
            color: 'rgba(0,0,0,0.70)'
          }
        },
        series: yData
      }
      this.chartOptions = options
    }
  }
}
</script>

<style lang="less" scoped>
.chartContent {
  width: 100%;
  height: 400px
}
</style>

# 内置图表-饼图

目前只有这一种类型的内置图表。

总数
基础用法
<template>
  <div class="chartContent">
    <SPieChart :data="chartData" :showPercentage="true" :pieInfoDesc="'总数'"></SPieChart>
  </div>
</template>

<script>
const colors = ['#9287E7', '#FF7B7B', '#FEB64D', '#49D6A5', '#32D3EB', '#F4605F', '#F49160', '#F2D43C', '#53CC77', '#36E4F7', '#58A7F2', '#717AF7', '#E24F74', '#C9EC75', '#78F8CB', '#9773EE']

let chartData = ['店外经营', '游摊小贩', '占道经营', '非机动车乱停放', '乱堆物堆料', '户外广告', '打包垃圾', '违规撑伞', '垃圾箱满溢', '暴露垃圾', '沿街晾晒'].map((item, index) => {
  return {
    'eventTypeName': item,
    'eventCount': randomNum(10),
    'eventRatio': '33.33'
  }
}).slice(0, 4).map((item, index) => {
  return {
    name: item.eventTypeName,
    num: item.eventCount,
    color: colors[index % colors.length]
  }
})

export default {
  data () {
    return {
      chartData: chartData
    }
  }
}
</script>

# API

# EChart

参数说明类型可选值默认值
optionsECharts图表的配置项Object—

# SPieChart

参数说明类型可选值默认值
data图表数据Array—[]
showPercentage显示百分比Boolean—false
disableEvent是否禁止响应鼠标事件Boolean—false
pieInfoDesc图表中间数字下面显示的文字String—

# SPieChart data选项字段

参数说明类型可选值默认值
name每项的名称String—
num每项的数量Number—
color该项在图表上显示的颜色String—
